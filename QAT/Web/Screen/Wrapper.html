def initialize name, options
  @name    = name
  @options = options
  parse_resolution options
  log.info "Created screen with #{width}x#{height}x#{depth}"
  @xvfb = nil
end

def destroy
  return unless @xvfb
  @xvfb.destroy
  log.info "Xvfb #{name} screen destroyed"
  @xvfb = nil
end

def start
  @xvfb = Headless.new @options
  @xvfb.start
  log.info "Xvfb #{name} screen started"
  @xvfb
end
